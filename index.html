          fillOpacity:0.08

        });

        const area=computeRectArea(bounds);

        aoiState={type:'extent',summary:'AOI: map extent',areaKm2:area};

        aoiSummaryChip.textContent=aoiState.summary;

        aoiAreaChip.textContent='Approx area: '+area.toFixed(2)+' km²';

        setAoiModeLabel('Map extent');

        setStatus('Current map extent captured as AOI.');

      }


      function applyRadiusAOI(){

        const lat=parseFloat(centerLatInput.value);

        const lng=parseFloat(centerLngInput.value);

        const r=parseFloat(radiusInput.value);

        if(!isFinite(lat)||!isFinite(lng)||!isFinite(r)){

          setStatus('Enter valid center and radius');

          return;

        }

        let km=r;

        if(radiusUnitSelect.value==='m')km=r/1000;

        if(radiusUnitSelect.value==='miles')km=r*1.60934;

        const meters=km*1000;

        clearAoi();

        const center={lat:lat,lng:lng};

        aoiOverlay=new google.maps.Circle({

          center:center,

          radius:meters,

          map:map,

          strokeColor:'#7fb0ff',

          strokeOpacity:0.95,

          strokeWeight:2,

          fillColor:'#3f7bff',

          fillOpacity:0.08

        });

        map.fitBounds(aoiOverlay.getBounds());

        const area=Math.PI*Math.pow(km,2);

        aoiState={type:'circle',summary:'AOI: point + radius',areaKm2:area};

        aoiSummaryChip.textContent=aoiState.summary;

        aoiAreaChip.textContent='Approx area: '+area.toFixed(2)+' km²';

        setAoiModeLabel('Point + radius');

      }


      function applyBBoxAOI(){

        const w=parseFloat(westInput.value);

        const s=parseFloat(southInput.value);

        const e=parseFloat(eastInput.value);

        const n=parseFloat(northInput.value);

        if(![w,s,e,n].every(v=>isFinite(v))){

          setStatus('Enter valid bounding box');

          return;

        }

        clearAoi();

        const bounds={west:w,south:s,east:e,north:n};

        aoiOverlay=new google.maps.Rectangle({

          bounds:bounds,

          map:map,

          strokeColor:'#7fb0ff',

          strokeOpacity:0.95,

          strokeWeight:2,

          fillColor:'#3f7bff',

          fillOpacity:0.08

        });

        map.fitBounds(bounds);

        const latKm=Math.abs(n-s)*111.32;

        const lngKm=Math.abs(e-w)*111.32*Math.cos(((n+s)/2)*Math.PI/180);

        const area=Math.max(latKm*lngKm,0);

        aoiState={type:'bbox',summary:'AOI: bounding box',areaKm2:area};

        aoiSummaryChip.textContent=aoiState.summary;

        aoiAreaChip.textContent='Approx area: '+area.toFixed(2)+' km²';

        setAoiModeLabel('Bounding box');

      }

      function loadScript(src) {

        return new Promise(function (resolve, reject) {

          const existing = document.querySelector('script[data-src="' + src + '"]');

          if (existing) {

            if (existing.dataset.loaded === 'true') {

              resolve();

              return;

            }

            existing.addEventListener('load', function () { resolve(); }, { once: true });

            existing.addEventListener('error', function () { reject(new Error('Failed to load script: ' + src)); }, { once: true });

            return;

          }


          const script = document.createElement('script');

          script.src = src;

          script.async = true;

          script.defer = true;

          script.dataset.src = src;

          script.onload = function () {

            script.dataset.loaded = 'true';

            resolve();

          };

          script.onerror = function () {

            reject(new Error('Failed to load script: ' + src));

          };

          document.head.appendChild(script);

        });

      }
